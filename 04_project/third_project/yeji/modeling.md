## DB modeling

```sql
-- classification 테이블
CREATE TABLE CLASSIFICATION(
		CLS_NUM		NUMBER				PRIMARY KEY NOT NULL
	,	CLS_NAME	VARCHAR2(30)	NOT NULL
	);

```

```sql
-- contact 테이블
CREATE TABLE CONTACT(
		PRI_NUM		NUMBER
	,	PRI_NAME	VARCHAR2(50)	NOT NULL
	,	PHONE_NUM	VARCHAR2(100)	NOT NULL
	,	EMAIL			VARCHAR2(100)	NOT NULL
	,	CLS_NUM		NUMBER
	,	CONSTRAINT 	CLS_CON_NUM PRIMARY KEY(PRI_NUM)
	,   CONSTRAINT	CLS_CLS_NUM	FOREIGN	KEY(CLS_NUM)
		REFERENCES	CLASSIFICATION(CLS_NUM)
		ON DELETE CASCADE
);
```
